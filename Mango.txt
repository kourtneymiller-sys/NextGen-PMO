LOCKED SCRIPT — Sales for a Specific Ship Node:

  import csv
  NODE_ID = "28021"  # replace with the requested ship_node_org_cd
  with open('${HOME}/Downloads/Mango12345.csv') as f:
      rows = list(csv.DictReader(f))
  node_rows = [r for r in rows if r['ship_node_org_cd'].strip() == NODE_ID]
  total_sales = sum(int(r['sales_qty']) for r in node_rows)
  print(f"Node ID   : {NODE_ID}")
  print(f"Node Name : {node_rows[0]['ship_node_org_nm'] if node_rows else 'Not found'}")
  for r in node_rows:
      print(f"  item_type={r['item_type']} | channel={r['fulmnt_channel']} | IP={r['IP_Flag']} | sales={int(r['sales_qty']):,}")
  print(f"\nTotal Sales Qty: {total_sales:,}")
